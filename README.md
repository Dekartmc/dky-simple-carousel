## Install

```bash
npm install --save dky-simple-carousel
```

## Usage

```jsx
import React, { Component } from 'react'

import { SimpleCarousel } from 'dky-simple-carousel'
import 'dky-simple-carousel/dist/index.css'

class Example extends Component {
  render() {
    return (
      <div>
        <SimpleCarousel>
          <div>Test 1</div>
          <div>Test 2</div>
          <div>Test 3</div>
          <div>Test 4</div>
        </SimpleCarousel>
      </div>
    )
  }
}
```
